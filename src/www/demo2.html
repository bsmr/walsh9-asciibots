                    For example, this page generates all the avatars from the usernames like this:
                    
$(".comment").each(function() { 
/* for each comment */
    var username = $(this).attr("data-username");
    /* get the username (here we us a data attribue) */
    var hash = CryptoJS.MD5(username).toString(CryptoJS.enc.Hex);
    /* take an md5 hash of it (using CryptoJS) */
    $(".bot", this).asciibot(hash.slice(-5));
    /* find the .bot element and apply the asciibot function, 
     * passing in the last 5 chars of the hash 
     */
});
